on run {input}	set theURL to input	tell application "Google Chrome"		if not (exists first window) then			make new window			set URL of last tab of first window to theURL		else			tell first window				set newTab to make new tab with properties {URL:theURL}			end tell		end if		activate	end tell	tell application "Safari"		close current tab of front window without saving		if (count of windows) > 0 then			activate		else			quit		end if	end tellend run
